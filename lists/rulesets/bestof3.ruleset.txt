<!ruleset-list>

name 'Best of Three'
require exact 2

alloc $1
alloc $2

match #match1 {
    white $1
    blue $2
}
match #match2 {
    white $1
    blue $2
}

order #match1, _clip, #match2

score {
    calc points among all
    resolve playoff

    first %1
    second %2
}

-- Fight Three
playoff {
    if equal at %2, %1
    
    match #match3 {
        white $1
        blue $2
    }
    
    first #match3.winner
    second #match3.loser
}
